# Generate Ed25519 SSH key (developer side)
ssh-keygen -t ed25519 -C "sovereign-dev@domain" -f ~/.ssh/id_ed25519

# Add public key to sovereign key registry (out-of-band, e.g., infra repo or Vault)
# Example registry file:
#   security/SOVEREIGN-ED25519-KEYS.txt

# Configure git to sign commits with SSH Ed25519
git config --global gpg.format ssh
git config --global user.signingkey ~/.ssh/id_ed25519.pub
git config --global commit.gpgsign true
